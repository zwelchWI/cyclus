00133 double pyne::endftod (char * s) {
00134   // Converts string from ENDF to float64.
00135   int pos, mant, exp;
00136   double v, dbl_exp;
00137 
00138   mant = exp = 0;
00139   if (s[2] == '.') {
00140     // Convert an ENDF float
00141     if (s[9] == '+' or s[9] == '-')
00142     {
00143       // All these factors of ten are from place values.
00144       mant = s[8] + 10 * s[7] + 100 * s[6] + 1000 * s[5] + 10000 * s[4] + \
00145              100000 * s[3] + 1000000 * s[1] - 1111111 * '0';
00146       exp = s[10] - '0';
00147       // Make the right power of 10.
00148       dbl_exp = exp & 01? 10.: 1;
00149       dbl_exp *= (exp >>= 1) & 01? 100.: 1;
00150       dbl_exp *= (exp >>= 1) & 01? 1.0e4: 1;
00151       dbl_exp *= (exp >>= 1) & 01? 1.0e8: 1;
00152       // Adjust for powers of ten from treating mantissa as an integer.
00153       dbl_exp = (s[9] == '-'? 1/dbl_exp: dbl_exp) * 1.0e-6;
00154       // Get mantissa sign, apply exponent.
00155       v = mant * (s[0] == '-'? -1: 1) * dbl_exp;
00156     }
00157     else {
00158       mant = s[7] + 10 * s[6] + 100 * s[5] + 1000 * s[4] + 10000 * s[3] + \
00159              100000 * s[1] - 111111 * '0';
00160       exp = s[10] + 10 * s[9] - 11 * '0';
00161       dbl_exp = exp & 01? 10.: 1;
00162       dbl_exp *= (exp >>= 1) & 01? 100.: 1;
00163       dbl_exp *= (exp >>= 1) & 01? 1.0e4: 1;
00164       dbl_exp *= (exp >>= 1) & 01? 1.0e8: 1;
00165       dbl_exp *= (exp >>= 1) & 01? 1.0e16: 1;
00166       dbl_exp *= (exp >>= 1) & 01? 1.0e32: 1;
00167       dbl_exp *= (exp >>= 1) & 01? 1.0e64: 1;
00168       dbl_exp = (s[8] == '-'? 1/dbl_exp: dbl_exp) * 1.0e-5;
00169       v = mant * (s[0] == '-'? -1: 1) * dbl_exp;
00170     }
00171   }
00172 
00173   // Convert an ENDF int to float; we start from the last char in the field and
00174   // move forward until we hit a non-digit.
00175   else {
00176     v = 0;
00177     mant = 1; // Here we use mant for the place value about to be read in.
00178     pos = 10;
00179     while (s[pos] != '-' and s[pos] != '+' and s[pos] != ' ' and pos > 0) {
00180       v += mant * (s[pos] - '0');
00181       mant *= 10;
00182       pos--;
00183     }
00184     v *= (s[pos] == '-'? -1: 1);
00185   }
00186   return v;
00187 }

00666 int pyne::nucname::id(int nuc)
00667 {
00668   if (nuc < 0)
00669     throw NotANuclide(nuc, "");
00670 
00671   int newnuc;
00672   int zzz = nuc / 10000000;     // ZZZ ?
00673   int aaassss = nuc % 10000000; // AAA-SSSS ?
00674   int aaa = aaassss / 10000;    // AAA ?
00675   // Nuclide must already be in id form
00676   if (0 < zzz && zzz <= aaa && aaa <= zzz * 7)
00677   {
00678     // Normal nuclide
00679     return nuc;
00680   }
00681   else if (aaassss == 0 && 0 < zz_name.count(zzz))
00682   {
00683     // Natural elemental nuclide:  ie for Urnaium = 920000000
00684     return nuc;
00685   }
00686   else if (nuc < 1000 && 0 < zz_name.count(nuc))
00687     //  Gave Z-number
00688     return nuc * 10000000;
00689 
00690   // Not in id form, try  ZZZAAAM form.
00691   zzz = nuc / 10000;     // ZZZ ?
00692   aaassss = nuc % 10000; // AAA-SSSS ?
00693   aaa = aaassss / 10;    // AAA ?
00694   if (zzz <= aaa && aaa <= zzz * 7)
00695   {
00696     // ZZZAAAM nuclide
00697     return (zzz*10000000) + (aaa*10000) + (nuc%10);
00698   }
00699   else if (aaa <= zzz && zzz <= aaa * 7 && 0 < zz_name.count(aaa))
00700   {
00701     // Cinder-form (aaazzzm), ie 2350920
00702     return (aaa*10000000) + (zzz*10000) + (nuc%10);
00703   }
00704   //else if (aaassss == 0 && 0 == zz_name.count(nuc/1000) && 0 < zz_name.count(zzz))
00705   else if (aaassss == 0 && 0 < zz_name.count(zzz))
00706   {
00707     // zzaaam form natural nuclide
00708     return zzz * 10000000;
00709   }
00710 
00711   if (nuc >= 1000000){
00712     // From now we assume no metastable info has been given.
00713     throw IndeterminateNuclideForm(nuc, "");
00714   };
00715 
00716   // Nuclide is not in zzaaam form, 
00717   // Try MCNP form, ie zzaaa
00718   // This is the same form as SZA for the 0th state.
00719   zzz = nuc / 1000;
00720   aaa = nuc % 1000; 
00721   if (zzz <= aaa)
00722   {
00723     if (aaa - 400 < 0)
00724     {
00725       if (nuc == 95242)
00726         return nuc * 10000 + 1;  // special case MCNP Am-242m
00727       else
00728         return nuc * 10000;  // Nuclide in normal MCNP form
00729     }
00730     else
00731     {
00732       // Nuclide in MCNP metastable form
00733       if (nuc == 95642)
00734         return (95642 - 400)*10000;  // special case MCNP Am-242
00735       nuc = ((nuc - 400) * 10000) + 1;
00736       while (3.0 < (float ((nuc/10000)%1000) / float (nuc/10000000)))
00737         nuc -= 999999;
00738       return nuc;
00739     }
00740   }
00741   else if (aaa == 0 && 0 < zz_name.count(zzz))
00742   {
00743     // MCNP form natural nuclide
00744     return zzz * 10000000;
00745   }
00746   else if (zzz > 1000)
00747   {
00748     // SZA form with a metastable state (sss != 0)
00749     int sss = zzz / 1000;
00750     int newzzz = zzz % 1000;
00751     
00752     return newzzz * 10000000 + aaa * 10000 + sss;
00753   }
00754 
00755   // Not a normal nuclide, might be a 
00756   // Natural elemental nuclide.  
00757   // ie 92 for Urnaium = 920000
00758   if (0 < zz_name.count(nuc))
00759     return nuc * 10000000;
00760   throw IndeterminateNuclideForm(nuc, "");

04312 void pyne::_load_atomic_mass_map()
04313 {
04314   // Loads the important parts of atomic_wight table into atomic_mass_map
04315 
04316   //Check to see if the file is in HDF5 format.
04317   if (!pyne::file_exists(pyne::NUC_DATA_PATH))
04318     throw pyne::FileNotFound(pyne::NUC_DATA_PATH);
04319 
04320   bool ish5 = H5Fis_hdf5(pyne::NUC_DATA_PATH.c_str());
04321   if (!ish5)
04322     throw h5wrap::FileNotHDF5(pyne::NUC_DATA_PATH);
04323 
04324   // Get the HDF5 compound type (table) description
04325   hid_t desc = H5Tcreate(H5T_COMPOUND, sizeof(atomic_mass_struct));
04326   H5Tinsert(desc, "nuc",   HOFFSET(atomic_mass_struct, nuc),   H5T_NATIVE_INT);
04327   H5Tinsert(desc, "mass",  HOFFSET(atomic_mass_struct, mass),  H5T_NATIVE_DOUBLE);
04328   H5Tinsert(desc, "error", HOFFSET(atomic_mass_struct, error), H5T_NATIVE_DOUBLE);
04329   H5Tinsert(desc, "abund", HOFFSET(atomic_mass_struct, abund), H5T_NATIVE_DOUBLE);
04330 
04331   // Open the HDF5 file
04332   hid_t nuc_data_h5 = H5Fopen(pyne::NUC_DATA_PATH.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
04333 
04334   // Open the data set
04335   hid_t atomic_mass_set = H5Dopen2(nuc_data_h5, "/atomic_mass", H5P_DEFAULT);
04336   hid_t atomic_mass_space = H5Dget_space(atomic_mass_set);
04337   int atomic_mass_length = H5Sget_simple_extent_npoints(atomic_mass_space);
04338 
04339   // Read in the data
04340   atomic_mass_struct * atomic_mass_array = new atomic_mass_struct[atomic_mass_length];
04341   H5Dread(atomic_mass_set, desc, H5S_ALL, H5S_ALL, H5P_DEFAULT, atomic_mass_array);
04342 
04343   // close the nuc_data library, before doing anythng stupid
04344   H5Dclose(atomic_mass_set);
04345   H5Fclose(nuc_data_h5);
04346 
04347   // Ok now that we have the array of stucts, put it in the map
04348   for(int n = 0; n < atomic_mass_length; n++){
04349     atomic_mass_map[atomic_mass_array[n].nuc] = atomic_mass_array[n].mass;
04350     natural_abund_map[atomic_mass_array[n].nuc] = atomic_mass_array[n].abund;
04351   }
04352 
04353   delete[] atomic_mass_array;
04354 };

04358 {
04359   // Find the nuclide's mass in AMU
04360   std::map<int, double>::iterator nuc_iter, nuc_end;
04361 
04362   nuc_iter = atomic_mass_map.find(nuc);
04363   nuc_end = atomic_mass_map.end();
04364 
04365   // First check if we already have the nuc mass in the map
04366   if (nuc_iter != nuc_end)
04367     return (*nuc_iter).second;
04368 
04369   // Next, fill up the map with values from the 
04370   // nuc_data.h5, if the map is empty.
04371   if (atomic_mass_map.empty())
04372   {
04373     // Don't fail if we can't load the library
04374     try
04375     {
04376       _load_atomic_mass_map();
04377       return atomic_mass(nuc);
04378     }
04379     catch(...){};
04380   };

04382   double aw;
04383   int nucid = nucname::id(nuc);

04388   {
04389     aw = atomic_mass((nucid/10000)*10000);
04390     atomic_mass_map[nuc] = aw;

04395   // take a best guess based on the 
04396   // aaa number.

04871   // Note that this may be expanded eventually to include other
04872   // sources of fission product data.
04873 
04874   // Find the parent/child pair branch ratio as a fraction
04875   if (source == 0) {
04876     std::map<std::pair<int, int>, double>::iterator fpy_iter, fpy_end;
04877     fpy_iter = wimsdfpy_data.find(from_to);

04879     if (fpy_iter != fpy_end)
04880         //if (get_error == true) return 0;
04881         return (*fpy_iter).second;
04882   } else {

04884     fpy_iter = ndsfpy_data.find(from_to);
04885     fpy_end = ndsfpy_data.end();
04886     if (fpy_iter != fpy_end) {
04887         switch (source) {
04888           case 1:
04889             if (get_error)
04890                 return (*fpy_iter).second.yield_thermal_err;
04891             return (*fpy_iter).second.yield_thermal;
04892             break;
04893           case 2:
04894             if (get_error)
04895                 return (*fpy_iter).second.yield_fast_err;
04896             return (*fpy_iter).second.yield_fast;
04897             break;
04898           case 3:
04899             if (get_error)
04900                 return (*fpy_iter).second.yield_14MeV_err;
04901             return (*fpy_iter).second.yield_14MeV;
04902             break;
04903         }
04904     }
04905   }
04906 
04907 
04908   // Next, fill up the map with values from the
04909   // nuc_data.h5, if the map is empty.
04910   if ((source == 0 ) && (wimsdfpy_data.empty())) {
