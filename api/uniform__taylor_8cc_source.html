00081   // terms, and the previous term Ck-1
00082   //

00084   // that Ck_sum slowly gets larger as more terms are added, rather than
00085   // simply multiplying a very small number by a very big one at the end

00091   int maxTerms = MaxNumTerms(alpha_t, tol);

00094   // number of terms has been reached

00097     C_next = (t / k) * B;
