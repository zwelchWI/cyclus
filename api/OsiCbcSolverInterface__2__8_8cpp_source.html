00550 {
00551   const double * lower = agentPtr_->solver()->getColLower();
00552   const double * upper = agentPtr_->solver()->getColUpper();
00553   const CoinPackedVector & lbs = cc.lbs();
00554   const CoinPackedVector & ubs = cc.ubs();
00555   int i;
00556 
00557   for ( i=0; i<lbs.getNumElements(); i++ ) {
00558     int iCol = lbs.getIndices()[i];
00559     double value = lbs.getElements()[i];
00560     if ( value > lower[iCol] )
00561       agentPtr_->solver()->setColLower(iCol, value);
00562   }
00563   for ( i=0; i<ubs.getNumElements(); i++ ) {
00564     int iCol = ubs.getIndices()[i];
00565     double value = ubs.getElements()[i];
00566     if ( value < upper[iCol] )
00567       agentPtr_->solver()->setColUpper(iCol, value);
