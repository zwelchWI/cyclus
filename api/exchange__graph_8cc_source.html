00049 double Capacity(ExchangeNode& n, const Arc& a) {

00051     throw cyclus::StateError("An notion of node capacity requires a nodegroup.");

00064     CLOG(cyclus::LEV_DEBUG1) << "    unit capacity: " << unit_caps[i];
00065     CLOG(cyclus::LEV_DEBUG1) << "         capacity: "
00066                              << group_caps[i] / unit_caps[i];
00067     
00068     // special case for unlimited capacities
00069     if (group_caps[i] == std::numeric_limits<double>::max()) {
00070       caps.push_back(std::numeric_limits<double>::max());

00072       caps.push_back(group_caps[i] / unit_caps[i]);

00074   }
